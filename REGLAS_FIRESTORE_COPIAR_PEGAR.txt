rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Reglas para la colección de usuarios
    match /usuarios/{userId} {
      // Permitir lectura al propio usuario autenticado
      allow read: if request.auth != null && request.auth.uid == userId;
      
      // Permitir escritura al propio usuario autenticado
      // Y también permitir actualización del campo premium desde el servidor (API routes)
      // Esto permite que el webhook de MercadoPago y el endpoint fixPremiumUser actualicen el estado premium
      allow write: if request.auth != null && request.auth.uid == userId
                   || (
                     // Permitir escritura si solo se están actualizando campos relacionados con premium
                     // y el campo premium está siendo establecido como true
                     request.resource.data.diff(resource.data).affectedKeys().hasOnly(['premium', 'premiumStatus', 'premiumSince', 'premiumPayment', 'updatedAt'])
                     && request.resource.data.premium == true
                     && (!resource.exists || resource.data.get('premium', false) == false)
                   );
      
      // Permitir creación del documento si es el propio usuario
      allow create: if request.auth != null && request.auth.uid == userId;
    }
    
    // Reglas para la colección de planes
    match /planes/{planId} {
      // Solo el usuario autenticado puede crear planes
      allow create: if request.auth != null && request.resource.data.userId == request.auth.uid;
      
      // Solo el dueño del plan puede leerlo
      allow read: if request.auth != null && resource.data.userId == request.auth.uid;
      
      // Solo el dueño del plan puede actualizarlo
      allow update: if request.auth != null && resource.data.userId == request.auth.uid;
      
      // Solo el dueño del plan puede eliminarlo
      allow delete: if request.auth != null && resource.data.userId == request.auth.uid;
    }
  }
}
